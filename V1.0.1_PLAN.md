# H5点餐系统 v1.0.1 优化计划

## 目标
修复数据库相关问题，提升系统稳定性和可靠性。

## 发现的潜在数据库问题

### 🔴 高优先级
1. **缺少事务处理**
   - 问题：创建订单时，如果中途失败可能导致数据不一致
   - 影响：订单数据完整性风险
   - 修复：使用 PostgreSQL 事务确保订单创建的原子性

2. **缺少连接池健康检查**
   - 问题：数据库连接断开后无法自动恢复
   - 影响：系统运行稳定性
   - 修复：添加连接池健康检查和自动重连机制

3. **缺少数据备份机制**
   - 问题：没有定期数据备份策略
   - 影响：数据丢失风险
   - 修复：添加数据库自动备份脚本

### 🟡 中优先级
4. **缺少数据库迁移系统**
   - 问题：数据库结构更新需要手动处理
   - 影响：版本升级困难
   - 修复：实现数据库迁移机制

5. **缺少查询性能监控**
   - 问题：无法识别慢查询
   - 影响：性能优化困难
   - 修复：添加查询日志和性能监控

6. **缺少数据验证约束**
   - 问题：应用层验证不够全面
   - 影响：可能插入不合理数据
   - 修复：在数据库层添加CHECK约束

### 🟢 低优先级
7. **缺少库存管理**
   - 问题：菜品售罄后仍可下单
   - 影响：用户体验问题
   - 修复：添加库存字段和售罄判断

8. **缺少数据清理机制**
   - 问题：历史订单数据持续增长
   - 影响：性能和存储成本
   - 修复：添加订单数据归档功能

## 修复计划

### Phase 1：核心稳定性修复（1-2天）
#### 1.1 添加事务处理
**文件**: `server/routes/orders.js`

```javascript
// 创建订单时使用事务
router.post('/', async (req, res) => {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    const { table_id, items, total_price, customer_phone, order_type } = req.body;

    // 验证数据
    if (!table_id || !items || items.length === 0 || !total_price) {
      throw new Error('订单信息不完整');
    }

    // 检查库存（如果添加了库存字段）
    for (const item of items) {
      const stockQuery = 'SELECT stock FROM dishes WHERE id = $1 FOR UPDATE';
      const stockResult = await client.query(stockQuery, [item.id]);

      if (stockResult.rows.length === 0 || stockResult.rows[0].stock < item.quantity) {
        throw new Error(`菜品 ${item.name} 库存不足`);
      }

      // 扣减库存
      await client.query(
        'UPDATE dishes SET stock = stock - $1 WHERE id = $2',
        [item.quantity, item.id]
      );
    }

    // 创建订单
    const createOrderQuery = `
      INSERT INTO orders (table_id, items, total_price, customer_phone, order_type)
      VALUES ($1, $2, $3, $4, $5)
      RETURNING *
    `;

    const values = [
      table_id,
      JSON.stringify(items),
      total_price,
      customer_phone || '',
      order_type !== undefined ? order_type : 0
    ];

    const result = await client.query(createOrderQuery, values);

    await client.query('COMMIT');

    res.json({ success: true, data: result.rows[0], message: '订单创建成功' });
  } catch (error) {
    await client.query('ROLLBACK');
    console.error('创建订单失败:', error);
    res.status(500).json({ success: false, message: error.message || '创建订单失败' });
  } finally {
    client.release();
  }
});
```

```javascript
// 添加健康检查函数
async function testConnection() {
  try {
    const client = await pool.connect();
    const result = await client.query('SELECT NOW()');
    client.release();
    console.log('🔍 数据库连接健康检查通过:', result.rows[0].now);
    return true;
  } catch (error) {
    console.error('❌ 数据库连接健康检查失败:', error);

  }
}

// 定期健康检查
setInterval(async () => {
  const isHealthy = await testConnection();
  if (!isHealthy) {

  }
}, 30000); // 每30秒检查一次

module.exports = { pool, testConnection };
```

#### 1.3 添加数据备份脚本
**新文件**: `scripts/backup-db.sh`

```bash
#!/bin/bash

# H5点餐系统 - 数据库备份脚本

# 配置
DB_NAME="${DB_NAME:-dining_mvp}"
DB_USER="${DB_USER:-dining_user}"
BACKUP_DIR="${BACKUP_DIR:-./backups}"
TIMESTAMP=$(date +"%Y%m%d_%H%M%S")
BACKUP_FILE="$BACKUP_DIR/dining_mvp_$TIMESTAMP.sql"
RETENTION_DAYS=7

# 创建备份目录
mkdir -p "$BACKUP_DIR"

# 备份数据库
echo "开始备份数据库: $DB_NAME"
pg_dump -U "$DB_USER" -h localhost "$DB_NAME" > "$BACKUP_FILE"

if [ $? -eq 0 ]; then
  echo "✅ 备份成功: $BACKUP_FILE"

  # 压缩备份文件
  gzip "$BACKUP_FILE"
  echo "✅ 压缩完成: $BACKUP_FILE.gz"

  # 删除旧备份
  find "$BACKUP_DIR" -name "dining_mvp_*.sql.gz" -mtime +$RETENTION_DAYS -delete
  echo "✅ 已删除 $RETENTION_DAYS 天前的旧备份"
else
  echo "❌ 备份失败"
  exit 1
fi
```

```sql
-- 菜品价格必须大于0
ALTER TABLE dishes ADD CONSTRAINT check_price_positive CHECK (price > 0);

-- 菜品状态只能是0或1
ALTER TABLE dishes ADD CONSTRAINT check_status_range CHECK (status IN (0, 1));

-- 订单总价必须大于0
ALTER TABLE orders ADD CONSTRAINT check_total_price_positive CHECK (total_price > 0);

-- 订单状态只能是'new'或'done'
ALTER TABLE orders ADD CONSTRAINT check_order_status CHECK (status IN ('new', 'done'));

-- 桌号不能为空
```

```javascript
// 包装查询方法以记录慢查询
const originalQuery = pool.query.bind(pool);
pool.query = function(text, params, callback) {
  const start = Date.now();

  return originalQuery(text, params, callback).then(result => {
    const duration = Date.now() - start;

    if (duration > 1000) { // 超过1秒的慢查询
      console.warn('⚠️  慢查询:', duration + 'ms');
      console.warn('SQL:', text);
      console.warn('参数:', params);
    }

    return result;
  });
};
```

```bash
#!/bin/bash

# 清理30天前的已完成订单
DAYS=30

psql -U "$DB_USER" -d "$DB_NAME" -c "
-- 创建归档表（如果不存在）
CREATE TABLE IF NOT EXISTS orders_archive AS
SELECT * FROM orders WHERE 1=0;

-- 归档旧订单
INSERT INTO orders_archive
SELECT * FROM orders
WHERE status = 'done' AND updated_at < NOW() - INTERVAL '$DAYS days';

-- 删除旧订单
DELETE FROM orders
WHERE status = 'done' AND updated_at < NOW() - INTERVAL '$DAYS days';

SELECT
  (SELECT COUNT(*) FROM orders_archive) as archived_orders,
  (SELECT COUNT(*) FROM orders WHERE status = 'done') as active_orders;
"
```

#### 4.2 添加健康检查API
**文件**: `server/routes/health.js`

```javascript
const router = require('express').Router();
const { testConnection } = require('../database');

// 健康检查接口
router.get('/', async (req, res) => {
  const health = {
    status: 'ok',
    timestamp: new Date().toISOString(),
    checks: {
      database: 'unknown',
      memory: 'unknown'
    }
  };

  // 数据库检查
  try {
    const dbHealthy = await testConnection();
    health.checks.database = dbHealthy ? 'healthy' : 'unhealthy';
  } catch (error) {
    health.checks.database = 'error';
    health.status = 'degraded';
  }

  // 内存检查
  const memoryUsage = process.memoryUsage();
  health.checks.memory = {
    heapUsed: Math.round(memoryUsage.heapUsed / 1024 / 1024) + ' MB',
    heapTotal: Math.round(memoryUsage.heapTotal / 1024 / 1024) + ' MB',
    external: Math.round(memoryUsage.external / 1024 / 1024) + ' MB'
  };

  // 如果有检查失败，返回503
  if (health.checks.database !== 'healthy') {
    return res.status(503).json(health);
  }

  res.json(health);
});

module.exports = router;
```
