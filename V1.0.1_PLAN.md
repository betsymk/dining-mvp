### 🔴 高优先级

#### 2.1 添加事务处理
**目标**: 确保订单创建的原子性和数据一致性

**文件修改**: `server/routes/orders.js`

**实现要点**:
```javascript
// 使用 PostgreSQL 事务
const client = await pool.connect();
try {
  await client.query('BEGIN');
  // 1. 验证订单数据
  // 2. 检查库存（如果添加了库存字段）
  // 3. 扣减库存
  // 4. 创建订单
  // 5. 提交事务
  await client.query('COMMIT');
} catch (error) {
  await client.query('ROLLBACK');
  throw error;
} finally {
  client.release();
}
```

- 订单创建失败时数据回滚
- 并发订单创建测试

**文件修改**: `server/database.js`

**实现要点**:
```javascript
// 健康检查函数
async function testConnection() {
  try {
    const client = await pool.connect();
    await client.query('SELECT NOW()');
    client.release();
    return true;
  } catch (error) {
    return false;
  }
}

// 定期检查（每30秒）
setInterval(async () => {
  const isHealthy = await testConnection();
  if (!isHealthy) {
    console.warn('数据库连接异常，尝试重新连接...');
  }
}, 30000);
```

#### 2.5 添加数据验证约束
**目标**: 在数据库层面防止不合理数据

**文件新增**: `migrations/add_constraints.sql`

**约束内容**:
```sql
-- 菜品价格必须大于0
ALTER TABLE dishes ADD CONSTRAINT check_price_positive CHECK (price > 0);

-- 菜品状态只能是0或1
ALTER TABLE dishes ADD CONSTRAINT check_status_range CHECK (status IN (0, 1));

-- 订单总价必须大于0
ALTER TABLE orders ADD CONSTRAINT check_total_price_positive CHECK (total_price > 0);

-- 订单状态只能是'new'或'done'
ALTER TABLE orders ADD CONSTRAINT check_order_status CHECK (status IN ('new', 'done'));

-- 桌号不能为空
ALTER TABLE orders ALTER COLUMN table_id SET NOT NULL;

-- 订单项不能为空
ALTER TABLE orders ALTER COLUMN items SET NOT NULL;
```

#### 2.6 添加库存管理
**目标**: 菜品售罄后不可下单

**文件新增**: `migrations/add_stock.sql`

**实现要点**:
```sql
-- 添加库存字段（默认100）
ALTER TABLE dishes ADD COLUMN stock INTEGER DEFAULT 100;

-- 库存非负约束
ALTER TABLE dishes ADD CONSTRAINT check_stock_non_negative CHECK (stock >= 0);

-- 添加库存索引
CREATE INDEX idx_dishes_stock ON dishes(stock) WHERE stock > 0;
```

**文件修改**: `client/index.html` - 前端库存显示和售罄标识

#### 2.7 添加查询性能监控
**目标**: 识别和优化慢查询

**文件修改**: `server/database.js`

**实现要点**:
```javascript
// 包装查询方法记录慢查询（>1秒）
const originalQuery = pool.query.bind(pool);
pool.query = function(text, params, callback) {
  const start = Date.now();
  return originalQuery(text, params, callback).then(result => {
    const duration = Date.now() - start;
    if (duration > 1000) {
      console.warn('⚠️  慢查询:', duration + 'ms');
      console.warn('SQL:', text);
    }
    return result;
  });
};
```

-- 优化订单按状态和时间查询
CREATE INDEX idx_orders_status_created

CREATE INDEX idx_dishes_category_status
ON dishes(category, status)
WHERE status = 1;

CREATE INDEX idx_orders_created_at_filter
ON orders(created_at DESC)

## 🚨 风险评估

### 高风险
- **事务处理影响性能**：需要评估事务对订单创建性能的影响
- **数据库迁移风险**：需要在测试环境充分验证
- **认证系统兼容性**：确保不影响现有API调用

### 中风险
- **备份脚本冲突**：需要确保备份不影响系统正常运行
- **健康检查误报**：需要合理设置阈值
- **库存管理复杂性**：需要考虑并发下单场景

### 低风险
- **日志文件增长**：需要配置日志轮转
- **归档数据访问**：需要确保归档数据可查询
- **索引空间占用**：监控索引大小
